103 private static IEnumerable<T> RandomSubsetImpl<T>(IEnumerable<T> source, Random rand, int? subsetSize)
104 {
105 // The simplest and most efficient way to return a random subset is to perform
106 // an in-place, partial Fisher-Yates shuffle of the sequence. While we could do
107 // a full shuffle, it would be wasteful in the cases where subsetSize is shorter
108 // than the length of the sequence.
109 // See: http://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle
110
111 T[] array = source.ToArray();
112 subsetSize ??= array.Length;
113
114 if (array.Length < subsetSize)
115 {
116 // TODO Throw InvalidOperationException instead?
117 throw new ArgumentOutOfRangeException(nameof(subsetSize),
118 "Subset size must be less than or equal to the source length.");
119 }
120
121 int m = 0; // keeps track of count items shuffled
122 int w = array.Length; // upper bound of shrinking swap range
123 int g = w - 1; // used to compute the second swap index
124
125 // perform in-place, partial Fisher-Yates shuffle
126 while (m < subsetSize)
127 {
128 int k = g - rand.Next(w);
129 (array[k], array[m]) = (array[m], array[k]);
130 ++m;
131 --w;
132 }
133
134 // yield the random subset as a new sequence
135 for (int i = 0; i < subsetSize; i++)
136 yield return array[i];
137 }
138 }
