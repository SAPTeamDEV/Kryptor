218 private void QueueProcessImpl()
219 {
220 int runningCount;
221 List<SessionHolder> waiting;
222
223 if (_sessionGroup != null)
224 {
225 runningCount = _sessionGroup.Running;
226 waiting = _sessionGroup.WaitingSessions;
227 }
228 else
229 {
230 runningCount = Holders.Where(x => x.Session.Status == SessionStatus.Running).Count();
231 waiting = Holders.Where(x => x.Session.Status == SessionStatus.NotStarted && SafeIsRady(x)).ToList();
232 }
233
234 if (waiting.Count == 0 || runningCount >= MaxRunningSessions) return;
235
236 int toBeStarted = Math.Min(MaxRunningSessions - runningCount, waiting.Count);
237 for (int i = 0; i < toBeStarted; i++)
238 {
239 try
240 {
241 SessionHolder sessionHolder = waiting[_sessionGroup != null ? 0 : Math.Min(waiting.Count - 1, i)];
242 StartManagedSession(sessionHolder);
243 }
244 catch
245 {
246 // Do not crash the SessionManager for ANY reasons. it may cause unexpected behavior and infinite loops.
247 if (_sessionHost.Verbose)
248 {
249 throw;
250 }
251 }
252 }
253 }
