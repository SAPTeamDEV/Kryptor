52 private byte[] CreateDynamicKey(CryptoProcess process)
53 {
54 byte[] key1 = KeyStore[Configuration.Continuous ? (process.ChunkIndex + process.BlockIndex) * 7 : (process.ChunkIndex - process.BlockIndex) * 3];
55 byte[] key2 = KeyStore[process.ChunkIndex - (Configuration.RemoveHash ? key1[4] : process.BlockHash[key1[27] % 32])];
56 byte[] key3 = KeyStore[process.ChunkIndex + process.BlockSize];

58 byte[] mKey = process.BlockIndex > 0

60 : key1.Concat(key2).Concat(key3).Concat(process.BlockHash).ToArray();
