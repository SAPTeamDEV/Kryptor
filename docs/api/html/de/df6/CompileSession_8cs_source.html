119 private async Task Compile(CancellationToken cancellationToken)
120 {
121 long lines = 0;
122 long words = 0;
123
124 using (StreamReader streamReader = new StreamReader(FilePath, Encoding.UTF8))
125 {
126 double steps = 1.0 / streamReader.BaseStream.Length * 100;
127
128 long readChars = 0;
129 string line;
130 while ((line = await AsyncCompat.ReadLineAsync(streamReader)) != null)
131 {
132 cancellationToken.ThrowIfCancellationRequested();
133 readChars += line.Length;
134 lines++;
135
136 Progress = readChars * steps;
137
138 line = line.Trim();
139
140 if (string.IsNullOrEmpty(line) || line.Length < 4 || !TryAddLine(line) || regex.IsMatch(line)) continue;
141
142 byte[] data = Encoding.UTF8.GetBytes(line + "\n");
143 words++;
144
145 if (!Indexing)
146 {
147 int c = WordlistFragmentCollection.GetWordIdentifier(line);
148 if (!fileStreams.ContainsKey(c))
149 {
150 lookupStrings[c] = line;
151 fileStreams[c] = File.Open(Path.Combine(DestPath, c.ToString()), FileMode.Create, FileAccess.ReadWrite);
152 }
153
154 fileStreams[c].Write(data, 0, data.Length);
155 }
156 }
157 }
158
159 IndexEntry.Words = words;
160 IndexEntry.Lines = lines;
161 }
