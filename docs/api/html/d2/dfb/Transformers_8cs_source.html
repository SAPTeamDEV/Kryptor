45 public static T[] Rotate<T>(T[] array, int positions)
46 {
47 int length = array.Length;
48 if (length == 0)
49 {
50 return Array.Empty<T>(); // Nothing to rotate
51 }
52
53 // Normalize the rotation amount (handle negative values and large rotations)
54 positions = ((positions % length) + length) % length;
55
56 // Create a temporary array to store rotated elements
57 T[] rotatedArray = new T[length];
58
59 // Copy elements after rotation to the temporary array
60 for (int i = 0; i < length; i++)
61 {
62 int newIndex = (i + positions) % length;
63 rotatedArray[newIndex] = array[i];
64 }
65

115 public static void Shuffle<T>(T[] array, int seed)
116 {
117 using (SHA256 sha256 = SHA256.Create())
118 {
119 int n = array.Length;
120
121 for (int i = n - 1; i > 0; i--)
122 {
123 // Compute a hash-based random index
124 byte[] hashBytes = sha256.ComputeHash(BitConverter.GetBytes(seed + i));
125 int j = Math.Abs(BitConverter.ToInt32(hashBytes, 0)) % (i + 1);
126
127 // Swap elements
128 (array[j], array[i]) = (array[i], array[j]);
129 }
130 }
131 }
