39 public static async Task<byte[]> EncryptAesEcbAsync(byte[] data, byte[] key, CancellationToken cancellationToken)
40 {
41 CheckArgument(data, nameof(data));
42 CheckArgument(key, nameof(key), 32);
43
44 using (Aes aesAlg = Aes.Create())
45 {
46 aesAlg.Key = key;
47 aesAlg.Mode = CipherMode.ECB; // Use ECB mode
48 aesAlg.Padding = PaddingMode.PKCS7;
49
50 ICryptoTransform encryptor = aesAlg.CreateEncryptor(aesAlg.Key, aesAlg.IV);
51
52 using (MemoryStream msEncrypt = new MemoryStream())
53 {
54 using (CryptoStream csEncrypt = new CryptoStream(msEncrypt, encryptor, CryptoStreamMode.Write))
55 {
56 await AsyncCompat.WriteAsync(csEncrypt, data, 0, data.Length, cancellationToken);
57 }
58
59 return msEncrypt.ToArray();
60 }
