83 public static T ReadHeader<T>(Stream stream)
84 where T : Header, new()
85 {
86 stream.Seek(0, SeekOrigin.Begin);
87
88 byte[] buffer = new byte[Math.Min(1024, stream.Length)];
89
90 int startPos = -1;
91 int endPos = -1;
92 int startIndex = 0;
93 int tries = 0;
94
95 while (true)
96 {
97 tries++;
98
99 // Get 1KB of data to search for header.
100 stream.Read(buffer, 0, buffer.Length);
101
102 if (startPos == -1)
103 {
104 int ts = buffer.LocatePattern(StartHeaderPattern);
105 if (ts != -1)
106 {
107 startPos = ts + StartHeaderPattern.Length;
108 startIndex = tries - 1;
109 }
110 }
111
112 if (startPos > -1 && endPos == -1)
113 {
114 int te = buffer.LocatePattern(EndHeaderPattern);
115 if (te != -1)
116 {
117 endPos = te;
118 }
119 }
120
121 if ((startPos == -1 && tries > 2) || (endPos == -1 && tries > 8 + startIndex))
122 {
123 stream.Seek(0, SeekOrigin.Begin);
124
125 return new T()
126 {
127 Verbosity = HeaderVerbosity.Empty
128 };
129 }
130
131 if (startPos > -1 && endPos > -1)
132 {
133 break;
134 }
135 }
136
137 stream.Seek(startPos, SeekOrigin.Begin);
138
139 byte[] dataBuffer = new byte[endPos - startPos];
140 stream.Read(dataBuffer, 0, dataBuffer.Length);
141
142 T header = ReadJson<T>(dataBuffer.Base64DecodeToString());
143
144 int detail = 0;
145 if (header.Version != null && header.EngineVersion != null)
146 {
147 detail++;
148
149 if (header.BlockSize > 0)
150 {
151 detail++;
152
153 if (header.Configuration != null)
154 {
155 detail++;
156 }
157 }
158 }
159
160 header.Verbosity = (HeaderVerbosity)detail;
161
162 stream.Seek(endPos + EndHeaderPattern.Length, SeekOrigin.Begin);
163
