187 void OnSessionProgressChanged(int slotId, object sender, SessionUpdateEventArgs e)
188 {
189 if (e.Progress <= 0 || e.Progress > 100) return;
190
191 var fragment = e.Progress / Count;
192 Interlocked.Exchange(ref _progress, _progress + fragment - _progressArray[slotId]);

204 else Unknown++;

206 ISession session = (ISession)sender;
207 Array.ForEach(e.Messages, (x) => messages.Add($"{session.Name} -> {x}"));

212 }
213
