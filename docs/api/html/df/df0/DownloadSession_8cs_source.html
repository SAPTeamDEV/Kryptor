98 private void SetEndStatus(object sender, System.ComponentModel.AsyncCompletedEventArgs e)
99 {
100 if (e.Error != null)
101 {
102 File.WriteAllText(PackageFile.FullName, JsonWorker.ToJson(Downloader.Package));
103 Exception = e.Error;
104 }
105 else
106 {
107 IndexEntry.Size = Downloader.Package.TotalFileSize;
108
109 Progress = -1;
110
111 Description = $"{IndexEntry.Id}: Verifying file";
112
113 VerifyHash(File.OpenRead(Downloader.Package.FileName), CancellationToken);
114
115 if (IndexEntry.Compressed)
116 {
117 Description = $"{IndexEntry.Id}: Extracting file";
118 string fileName = new Request(IndexEntry.Uri.ToString()).GetFileName().Result;
119
120 if (fileName.ToLower().EndsWith(".7z"))
121 {
122 SevenZipArchive reader = SevenZipArchive.Open(Downloader.Package.FileName);
123 reader.Entries.First().WriteToFile(OutputFile.FullName);
124 reader.Dispose();
125 }
126 else
127 {
128 IReader reader = ReaderFactory.Open(File.OpenRead(Downloader.Package.FileName));
129 reader.MoveToNextEntry();
130 reader.WriteEntryTo(OutputFile);
131 reader.Dispose();
132 }
133 }
134 else
135 {
136 if (File.Exists(OutputFile.FullName))
137 {
138 OutputFile.Delete();
139 }
140
141 File.Move(Downloader.Package.FileName, OutputFile.FullName);
142 }
143
144 if (File.Exists(Downloader.Package.FileName))
145 {
146 File.Delete(Downloader.Package.FileName);
147 }
148
149 Progress = 100;
150
151 Description = $"{IndexEntry.Id}: Download completed";
152 }

165 protected override async Task<bool> RunAsync(ISessionHost sessionHost, CancellationToken cancellationToken)
166 {
167 CancellationToken = cancellationToken;
168 Description = $"{IndexEntry.Id}: Starting download";
169
170 if (File.Exists(PackageFile.FullName))
171 {
172 DownloadPackage package = JsonWorker.ReadJson<DownloadPackage>(File.ReadAllText(PackageFile.FullName));
173 PackageFile.Delete();
174
175 await Downloader.DownloadFileTaskAsync(package, cancellationToken);
176 }
177 else
178 {
179 var dest = Utilities.EnsureDirectoryExists(Path.Combine(Path.GetTempPath(), Path.GetRandomFileName()));
180
181 var tempFile = Path.Combine(dest, HashString);
182
183 await Downloader.DownloadFileTaskAsync(IndexEntry.Uri.OriginalString, tempFile, cancellationToken);
184 }
185
186 Downloader.Dispose();
187
188 return Exception != null ? throw Exception : true;
189 }

204 private void VerifyHash(Stream stream, CancellationToken cancellationToken)
205 {
206 try
207 {
208 byte[] hash = stream.Sha256();
209
210 if (IndexEntry.Hash == null)
211 {
212 IndexEntry.Hash = hash;
213 }
214 else
215 {
216 if (!IndexEntry.Hash.SequenceEqual(hash))
217 {
218 throw new InvalidDataException("File is corrupted");
219 }
220 }
221 }
222 catch (InvalidDataException)
223 {
224 stream?.Dispose();
225 throw;
226 }
227 catch
228 {
229 // Ignore hash errors
230 }
231 finally
232 {
233 stream?.Dispose();
234 }
235 }
