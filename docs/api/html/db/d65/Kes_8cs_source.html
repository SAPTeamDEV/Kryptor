234 private async Task ProcessDataAsync(Stream source, Stream dest, bool doEncrypt, CancellationToken cancellationToken)
235 {
236 CryptoProcess process = new CryptoProcess();
237 process.InitializeData();
238
239 Func<CryptoProcess, int> blockSizeCallback;
240 Func<byte[], CryptoProcess, Task<byte[]>> cryptoCallback;
241 int chunckSize;
242
243 if (doEncrypt)
244 {
245 blockSizeCallback = GetEncryptionBufferSize;
246 cryptoCallback = EncryptBlockAsync;
247 chunckSize = Provider.EncryptionChunkSize;
248 }
249 else
250 {
251 blockSizeCallback = GetDecryptionBufferSize;
252 cryptoCallback = DecryptBlockAsync;
253 chunckSize = Provider.DecryptionChunkSize;
254 }
255
256 double step = (double)((double)chunckSize / (source.Length - source.Position)) * 100;
257 int counter = 0;
258 int blockSize;
259 long i = 0;
260 ProgressChanged?.Invoke(this, 0);
261
262 try
263 {
264 while (i < source.Length)
265 {
266 process.BlockSize = Provider.Configuration.DynamicBlockProccessing ? DynamicEncryption.GetDynamicBlockSize(Provider.KeyStore, process) : BlockSize;
267 blockSize = blockSizeCallback(process);
268 int actualSize = (int)Math.Min(source.Length - source.Position, blockSize);
269
270 byte[] slice = new byte[actualSize];
271 await AsyncCompat.ReadAsync(source, slice, 0, slice.Length, cancellationToken);
272 byte[] eSlice = await cryptoCallback(slice, process);
273 await AsyncCompat.WriteAsync(dest, eSlice, 0, eSlice.Length, cancellationToken);
274
275 counter += blockSize / chunckSize;
276 double prog = step * counter;
277 ProgressChanged?.Invoke(this, Math.Min(prog, 100));
278
279 process.NextBlock(!Provider.Configuration.Continuous);
280 i += blockSize;
281 }
282 }
283 catch (ArgumentOutOfRangeException)
284 {
285 throw new DataException("Small keystore size");
286 }
287 }
