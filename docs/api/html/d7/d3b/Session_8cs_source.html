186 public async Task StartAsync(ISessionHost sessionHost, CancellationToken cancellationToken)
187 {
188 ThrowIfEnded();
189
190 try
191 {
192 cancellationToken.ThrowIfCancellationRequested();
193
194 if (IsRunning)
195 {
196 throw new InvalidOperationException("The session is already running");
197 }
198
199 if (!IsReady(cancellationToken))
200 {
201 throw new InvalidOperationException("You may not start this session at the moment");
202 }
203
204 Status = SessionStatus.Running;
205 Timer.Start();
206
207 bool result = await RunAsync(sessionHost, cancellationToken);
208
209 if (result)
210 {
211 EndReason = SessionEndReason.Completed;
212 }
213 }
214 catch (OperationCanceledException ocex)
215 {
216 EndReason = cancellationToken.IsCancellationRequested ? SessionEndReason.Cancelled : SessionEndReason.Failed;
217 Exception = ocex;
218 }
219 catch (Exception ex)
220 {
221 if (ex is AggregateException aex && aex.InnerException != null)
222 {
223 ex = ex.InnerException;
224 }
225
226 EndReason = SessionEndReason.Failed;
227 Exception = ex;
228 }
229 finally
230 {
231 if (EndReason == SessionEndReason.Failed)
232 {
233 SetErrorMessage(sessionHost, Exception);
234 }
235
236 Timer.Stop();
237 Status = SessionStatus.Ended;
238 }
