40 public static void RegisterProvider<T>(string prefix, string displayName, params string[] hints)
41 where T : CryptoProvider, new()
42 {
43 if (!allowKryptorPrefix && prefix == "kryptor")
44 {
45 throw new ArgumentException("The kryptor prefix is only allowed for internal crypto providers");
46 }
47
48 CryptoProvider instance = new T();
49 string name = $"{prefix}:{instance.GetType().Name}";
50
51 if (!GlobalProviders.ContainsKey(name))
52 {
53 GlobalProviders[name] = (displayName, instance);
54
55 foreach (string hint in hints)
56 {
57 string hintName = $"{prefix}:{hint}";
58 if (!GlobalHints.ContainsKey(hintName))
59 {
60 GlobalHints.Add(hintName, name);
61 }
62 else
63 {
64 throw new ArgumentException("A provider with this hint already registered: " + hintName);
65 }
66 }
67 }
68 else
69 {
70 throw new ArgumentException("A provider with this name already registered: " + name);
71 }
